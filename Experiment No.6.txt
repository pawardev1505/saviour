Experiment No.6


Write a program to apply validations to Registration Form



index.html


<!DOCTYPE html> 
<html> 
<head> 
    <title>Form in HTML</title>
    <link rel="stylesheet" type="text/css" href="styles.css"> 
    <script>
        function val() {
            let sname = document.forms["f1"]["sname"].value;
            let email = document.f1.email.value;
            let pwd = document.f1.pwd.value;
            let cpwd = document.f1.cpwd.value;
            let gender = document.f1.gender;

            if (!(isNaN(sname))) {
                alert("Enter only characters for name");
                document.f1.sname.focus();
                return false;
            }
            if (pwd.length < 8 || cpwd.length < 8) {
                alert("Password must be a minimum of 8 characters in length");
                document.f1.pwd.focus();
                return false;
            }
            if (pwd != cpwd) {
                alert("Passwords do not match");
                document.f1.pwd.focus();
                return false;
            }
            let genderSelected = false;
            for (let i = 0; i < gender.length; i++) {
                if (gender[i].checked) {
                    genderSelected = true;
                    break;
                }
            }
            if (!genderSelected) {
                alert("Please choose your gender");
                return false;
            }

            return true; // All validations passed
        }
    </script> 
</head> 
<body align="center"> 
    <h2>Registration Form</h2> 
    <form method="get" name="f1" class="ff" onsubmit="return val();"> 
        <fieldset> 
            <legend>User Personal Information</legend> 
            <label for="sname">Enter your full name</label><br> 
            <input type="text" id="sname" name="sname" required><br> 
            
            <label for="email">Enter your email</label><br> 
            <input type="email" id="email" name="email" required><br> 
            
            <label for="pwd">Enter your password</label><br> 
            <input type="password" id="pwd" name="pwd" required><br> 
            
            <label for="cpwd">Confirm your password</label><br> 
            <input type="password" id="cpwd" name="cpwd" required><br> 
            
            <br><label>Enter your gender</label><br> 
            <input type="radio" name="gender" value="male" required> Male <br> 
            <input type="radio" name="gender" value="female"> Female <br> 
            <input type="radio" name="gender" value="others"> Others <br> 
            
            <br>Enter your address:<br> 
            <textarea name="address" required></textarea><br> 
            
            <input type="submit" value="SUBMIT"> 
        </fieldset> 
    </form> 
</body> 
</html>
