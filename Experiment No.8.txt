
Experiment No.8

Write a program to demonstrate Class Component & States using React JS



npx create-react-app my-app
cd my-app

Replace the contents of the generated src folder with your code files 
(App.js, App1.js, Classcomp.js, indexclass.js, etc.).
Also, make sure your index.js is set up correctly to render the main component.


index.js (main file)

import React from 'react';
import ReactDOM from 'react-dom/client'; // Note: use 'react-dom/client' for React 18+
import './index.css';
import App from './App'; // or App1, depending on which component you want to start with

const root = ReactDOM.createRoot(document.getElementById('root'));
root.render(
    <React.StrictMode>
        <App /> {/* or <App1 /> if you want to start with App1 */}
    </React.StrictMode>
);



App.js

import { Component } from "react";

    return (

export default App;

Classcomp.js


import { Component } from "react";
import './index.css';

class Classcomp extends Component {
    render() {
        const { color } = this.props;
        return (
            <div>
                <h1 style={{ color }}>This is the first class component</h1>
                <h3>Demo of class component</h3>
                <ul className="l1">
                    <li>IP</li>
                    <li>CNS</li>
                    <li>EEB</li>
                </ul>
            </div>
        );
    }
}

export default Classcomp;
